433 void EnsureValidity(std::vector<int> origin, std::vector<int> size, std::string component, std::string id) {
434 if (origin.at(0) < -8 ||
435 origin.at(0) > 8 ||
436 origin.at(1) < 0 ||
437 origin.at(1) > 16 ||
438 origin.at(2) < -8 ||
439 origin.at(2) > 8) {
440 adk::log::error(
441 "{} - Minimal position of the bounds of the {} box. "
442 "\"origin\" "
443 "is "
444 "specified as [x, y, z] and must be in the range (-8, "
445 "0, "
446 "-8) "
447 "to (8, 16, 8), inclusive",
448 id, component);
449 exit(EXIT_FAILURE);
450 }
451
452 int origin_1 = origin.at(0);
453 int size_1 = size.at(0);
454 int origin_2 = origin.at(1);
455 int size_2 = size.at(1);
456 int origin_3 = origin.at(2);
457 int size_3 = size.at(2);
458
459 if ((origin_1 + size_1) < -8 || (origin_1 + size_1) > 8) {
460 adk::log::error(
461 "{} - Size of each side of the {} box. Size is "
462 "specified as [x, y, z]. \"origin\" + \"size\" must be "
463 "in the range (-8, 0, -8) to (8, 16, 8), inclusive",
464 id, component);
465 exit(EXIT_FAILURE);
466 }
467
468 if ((origin_2 + size_2) < 0 || (origin_2 + size_2) > 16) {
469 adk::log::error(
470 "{} - Size of each side of the {} box. Size is "
471 "specified as [x, y, z]. \"origin\" + \"size\" must be "
472 "in the range (-8, 0, -8) to (8, 16, 8), inclusive",
473 id, component);
474 exit(EXIT_FAILURE);
475 }
476
477 if ((origin_3 + size_3) < -8 || (origin_3 + size_3) > 8) {
478 adk::log::error(
479 "{} - Size of each side of the {} box. Size is "
