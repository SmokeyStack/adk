361
362 struct BlockPlacementFilter {
363 std::vector<BlockAllowedFaces> allowed_faces;
364 std::vector<std::variant<std::string, BlockDescriptor>> block_filter;
365 };
366
367 void to_json(nlohmann::json& j, const BlockPlacementFilter& p) {
368 std::vector<std::string> allowed_faces;
369 for each (BlockAllowedFaces var in p.allowed_faces) {
370 allowed_faces.push_back(GetBlockAllowedFaces(var));
371 }
372 nlohmann::json j2 = nlohmann::json::array();
373 for each (auto entries in p.block_filter) {
374 if (std::holds_alternative<std::string>(entries)) {
375 j2.push_back(std::get<std::string>(entries));
376 }
377 else {
378 j2.push_back(std::get<BlockDescriptor>(entries));
379 }
380 }
381
382 j = nlohmann::json{ {"allowed_faces", allowed_faces}, {"block_filter", j2} };
383 }
