87 struct BlockPlacementFilter {
88 std::vector<BlockAllowedFaces> allowed_faces;
89 std::vector<std::variant<std::string, BlockDescriptor>> block_filter;
90 };
91
92 void to_json(nlohmann::json& j, const BlockPlacementFilter& p) {
93 std::vector<std::string> allowed_faces;
94 for each (BlockAllowedFaces var in p.allowed_faces) {
95 allowed_faces.push_back(GetBlockAllowedFaces(var));
96 }
97 nlohmann::json j2 = nlohmann::json::array();
98 for each (auto entries in p.block_filter) {
99 if (std::holds_alternative<std::string>(entries)) {
100 j2.push_back(std::get<std::string>(entries));
101 }
102 else {
103 j2.push_back(std::get<BlockDescriptor>(entries));
104 }
105 }
106
107 j = nlohmann::json{ {"allowed_faces", allowed_faces}, {"block_filter", j2} };
108 }
