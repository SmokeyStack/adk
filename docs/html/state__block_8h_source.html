47nlohmann::json::object_t TranslateMaterialInstances(adk::BlockTextures textures) {
48 nlohmann::json temp, output;
49 temp = textures.all;
50 if (!temp.is_null())
51 output["minecraft:material_instances"]["*"] = temp;
52 temp = textures.ends;
53 if (!temp.is_null()) {
54 output["minecraft:material_instances"]["ends"] = temp;
55 output["minecraft:material_instances"]["up"] = "ends";
56 output["minecraft:material_instances"]["down"] = "ends";
57 }
58 temp = textures.sides;
59 if (!temp.is_null()) {
60 output["minecraft:material_instances"]["sides"] = temp;
61 output["minecraft:material_instances"]["north"] = "sides";
62 output["minecraft:material_instances"]["east"] = "sides";
63 output["minecraft:material_instances"]["south"] = "sides";
64 output["minecraft:material_instances"]["west"] = "sides";
65 }
66 temp = textures.up;
67 if (!temp.is_null())
68 output["minecraft:material_instances"]["up"] = temp;
69 temp = textures.down;
70 if (!temp.is_null())
71 output["minecraft:material_instances"]["down"] = temp;
72 temp = textures.north;
73 if (!temp.is_null())
74 output["minecraft:material_instances"]["north"] = temp;
75 temp = textures.east;
76 if (!temp.is_null())
77 output["minecraft:material_instances"]["east"] = temp;
78 temp = textures.south;
79 if (!temp.is_null())
80 output["minecraft:material_instances"]["up"] = temp;
81 temp = textures.west;
82 if (!temp.is_null())
83 output["minecraft:material_instances"]["west"] = temp;
84
85 return output;
86}
